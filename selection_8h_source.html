746  template<class F>
747  void split(std::vector<Selection>& parts, F callback){
748  using namespace std;
749  parts.clear();
750  using Ret = decltype(callback(*this,0));
751  // Map
752  map<Ret,vector<int> > m;
753  for(int i=0; i<size(); ++i){
754  m[callback(*this,i)].push_back(Index(i));
755  }
756  // Create selections
757  typename map<Ret,vector<int> >::iterator it;
758  for(it=m.begin();it!=m.end();it++){
759  parts.push_back(Selection(*system));
760  parts.back().modify( it->second.begin(), it->second.end() );
761  }
762  }
