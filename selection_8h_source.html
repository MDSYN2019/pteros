768  void split(std::vector<Selection>& parts, F callback){
769  using namespace std;
770  parts.clear();
771  using Ret = decltype(callback(*this,0));
772  // Map
773  map<Ret,vector<int> > m;
774  for(int i=0; i<size(); ++i){
775  m[callback(*this,i)].push_back(index(i));
776  }
777  // Create selections
778  typename map<Ret,vector<int> >::iterator it;
779  for(it=m.begin();it!=m.end();it++){
780  parts.push_back(Selection(*system));
781  parts.back().modify( it->second.begin(), it->second.end() );
782  }
783  }
