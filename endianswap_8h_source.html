32 static void swap2_unaligned(void *v, long ndata) {
33  long i;
34  char * dataptr = (char *) v;
35  char tmp;
36 
37  for (i = 0; i < ndata-1; i += 2) {
38  tmp = dataptr[i];

41  }
42 }

65 static void swap8_unaligned(void *v, long ndata) {
66  char *data = (char *) v;
67  long i;
68  char byteArray[8];
69  char *bytePointer;
70 
71  for (i=0; i<ndata; i++) {
72  bytePointer = data + (i<<3);
73  byteArray[0] = *bytePointer;
74  byteArray[1] = *(bytePointer+1);
75  byteArray[2] = *(bytePointer+2);
76  byteArray[3] = *(bytePointer+3);
77  byteArray[4] = *(bytePointer+4);
78  byteArray[5] = *(bytePointer+5);
79  byteArray[6] = *(bytePointer+6);
80  byteArray[7] = *(bytePointer+7);

84  *(bytePointer+2) = byteArray[5];

122 /* Only works with aligned 8-byte quantities, will cause a bus error */
123 /* on some platforms if used on unaligned data. */
124 static void swap8_aligned(void *v, long ndata) {
125  /* Use int* internally to prevent bugs caused by some compilers */
126  /* and hardware that would potentially load data into an FP reg */
127  /* and hose everything, such as the old "jmemcpy()" bug in NAMD */
128  int *data = (int *) v;
129  long i;
130  int *N;
131  int t0, t1;
132 
133  for (i=0; i<ndata; i++) {
134  N = data + (i<<1);
135  t0 = N[0];
136  t0=(((t0>>24)&0xff) | ((t0&0xff)<<24) |
137  ((t0>>8)&0xff00) | ((t0&0xff00)<<8));
138 
139  t1 = N[1];
140  t1=(((t1>>24)&0xff) | ((t1&0xff)<<24) |
141  ((t1>>8)&0xff00) | ((t1&0xff00)<<8));
142 
143  N[0] = t1;
144  N[1] = t0;
145  }
146 }
