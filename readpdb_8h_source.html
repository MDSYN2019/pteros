35 /* read the next record from the specified pdb file, and put the string found
36  in the given string pointer (the caller must provide adequate (81 chars)
37  buffer space); return the type of record found
38 */
39 static int read_pdb_record(FILE *f, char *retStr) {
40  int ch;
41  char inbuf[PDB_BUFFER_LENGTH]; /* space for line + cr + lf + NUL */
42  int recType = PDB_UNKNOWN;
43 
44  /* XXX This PDB record reading code breaks with files that use
45  * Mac or DOS style line breaks with ctrl-M characters. We need
46  * to replace the use of fgets() and comparisons against \n with
47  * code that properly handles the other cases.
48  */
49 
50  /* read the next line, including any ending cr/lf char */
51  if (inbuf != fgets(inbuf, PDB_RECORD_LENGTH + 2, f)) {
52  retStr[0] = '\0';
53  recType = PDB_EOF;
54  } else {
55 #if 0
56  /* XXX disabled this code since \n chars are desirable in remarks */
57  /* and to make the behavior consistent with webpdbplugin */
58 
59  /* remove the newline character, if there is one */
60  if (inbuf[strlen(inbuf)-1] == '\n')
61  inbuf[strlen(inbuf)-1] = '\0';
62 #endif
63 
64  /* atom records are the most common */
65  if (!strncmp(inbuf, "ATOM ", 5) || !strncmp(inbuf, "HETATM", 6)) {
66  /* Note that by only comparing 5 chars for "ATOM " rather than 6, */
67  /* we allow PDB files containing > 99,999 atoms generated by AMBER */
68  /* to load which would otherwise fail. Not needed for HETATM since */
69  /* those aren't going to show up in files produced for/by MD engines. */
70  recType = PDB_ATOM;
71  } else if (!strncmp(inbuf, "CONECT", 6)) {
72  recType = PDB_CONECT;
73  } else if (!strncmp(inbuf, "REMARK", 6)) {
74  recType = PDB_REMARK;
75  } else if (!strncmp(inbuf, "CRYST1", 6)) {
76  recType = PDB_CRYST1;
77  } else if (!strncmp(inbuf, "HEADER", 6)) {
78  recType = PDB_HEADER;
79  } else if (!strncmp(inbuf, "END", 3)) { /* very permissive */
80  /* XXX we treat any "ENDxxx" record as an end, to simplify testing */
81  /* since we don't remove trailing '\n' chars */
82 
83  /* the only two legal END records are "END " and "ENDMDL" */
84  recType = PDB_END;
85  }
86 
87 #if 0
88  /* XXX disable record type checking for now */
89  if (recType == PDB_ATOM ||
90  recType == PDB_CONECT ||
91  recType == PDB_REMARK ||
92  recType == PDB_HEADER ||
93  recType == PDB_CRYST1) {
94  strcpy(retStr, inbuf);
95  } else {
96  retStr[0] = '\0';
97  }
98 #else
99  strcpy(retStr, inbuf);
100 #endif
101  }
102 
103  /* read the '\r', if there was one */
104  ch = fgetc(f);
105  if (ch != '\r')
106  ungetc(ch, f);
107 
108  return recType;
109 }

112 /* Extract the alpha/beta/gamma a/b/c unit cell info from a CRYST1 record */
113 static void get_pdb_cryst1(const char *record,
114  float *alpha, float *beta, float *gamma,
115  float *a, float *b, float *c) {
116  char tmp[PDB_RECORD_LENGTH+3]; /* space for line + cr + lf + NUL */
117  char ch, *s;
118  memset(tmp, 0, sizeof(tmp));
119  strncpy(tmp, record, PDB_RECORD_LENGTH);
120 
121  s = tmp+6 ; ch = tmp[15]; tmp[15] = 0;
122  *a = (float) atof(s);
123  s = tmp+15; *s = ch; ch = tmp[24]; tmp[24] = 0;
124  *b = (float) atof(s);
125  s = tmp+24; *s = ch; ch = tmp[33]; tmp[33] = 0;
126  *c = (float) atof(s);
127  s = tmp+33; *s = ch; ch = tmp[40]; tmp[40] = 0;
128  *alpha = (float) atof(s);
129  s = tmp+40; *s = ch; ch = tmp[47]; tmp[47] = 0;
130  *beta = (float) atof(s);
131  s = tmp+47; *s = ch; ch = tmp[54]; tmp[54] = 0;
132  *gamma = (float) atof(s);
133 }

170 /* remove leading and trailing spaces from PDB fields */

173 
174  len = strlen(field);

207 static void get_pdb_conect(const char *record, int natoms, int *idxmap,
208  int *maxbnum, int *nbonds, int **from, int **to) {
209  int bondto[11], numbonds, i;
210 
211  int reclen = strlen(record);
212  for (numbonds=0, i=0; i<11; i++) {
213  char bondstr[6];
214  const int fieldwidth = 5;
215  int start = 6 + i*fieldwidth;
216  int end = start + fieldwidth;
217 
218  if (end >= reclen)
219  break;
220 
221  memcpy(bondstr, record + start, fieldwidth);
222  bondstr[5] = '\0';
223  if (sscanf(bondstr, "%d", &bondto[numbonds]) < 0)
224  break;
225  numbonds++;
226  }
227 
228  for (i=0; i<numbonds; i++) {
229  /* only add one bond per pair, PDBs list them redundantly */
230  if (bondto[i] > bondto[0]) {
231  int newnbonds = *nbonds + 1; /* add a new bond */
232 
233  /* allocate more bondlist space if necessary */
234  if (newnbonds >= *maxbnum) {
235  int newmax;
236  int *newfromlist, *newtolist;
237  newmax = (newnbonds + 11) * 1.25;
238 
239  newfromlist = (int *) realloc(*from, newmax * sizeof(int));
240  newtolist = (int *) realloc(*to, newmax * sizeof(int));
241 
242  if (newfromlist != NULL || newtolist != NULL) {
243  *maxbnum = newmax;
244  *from = newfromlist;
245  *to = newtolist;
246  } else {
247  printf("readpdb) failed to allocate memory for bondlists\n");
248  return; /* abort */
249  }
250  }
251 
252  *nbonds = newnbonds;
253  (*from)[newnbonds-1] = idxmap[bondto[0]] + 1;
254  (*to)[newnbonds-1] = idxmap[bondto[i]] + 1;
255  }
256  }
257 }

347 /* Write PDB data to given file descriptor; return success. */
348 static int write_raw_pdb_record(FILE *fd, const char *recordname,
349  int index,const char *atomname, const char *resname,int resid,
350  const char *insertion, const char *altloc, const char *elementsymbol,
351  float x, float y, float z, float occ, float beta,
352  const char *chain, const char *segname) {
353  int rc;
354  char indexbuf[32];
355  char residbuf[32];
356  char segnamebuf[5];
357  char resnamebuf[5];
358  char altlocchar;
359 
360  /* XXX */
361  /* if the atom or residue indices exceed the legal PDB spec, we */
362  /* start emitting asterisks or hexadecimal strings rather than */
363  /* aborting. This is not really legal, but is an accepted hack */
364  /* among various other programs that deal with large PDB files */
365  /* If we run out of hexadecimal indices, then we just print */
366  /* asterisks. */
367  if (index < 100000) {
368  sprintf(indexbuf, "%5d", index);
369  } else if (index < 1048576) {
370  sprintf(indexbuf, "%05x", index);
371  } else {
372  sprintf(indexbuf, "*****");
373  }
374 
375  if (resid < 10000) {
376  sprintf(residbuf, "%4d", resid);
377  } else if (resid < 65536) {
378  sprintf(residbuf, "%04x", resid);
379  } else {
380  sprintf(residbuf, "****");
381  }
382 
383  altlocchar = altloc[0];
384  if (altlocchar == '\0') {
385  altlocchar = ' ';
386  }
387 
388  /* make sure the segname or resname do not overflow the format */
389  strncpy(segnamebuf,segname,4);
390  segnamebuf[4] = '\0';
391  strncpy(resnamebuf,resname,4);
392  resnamebuf[4] = '\0';
393 
394 
395  rc = fprintf(fd,
396  "%-6s%5s %4s%c%-4s%c%4s%c %8.3f%8.3f%8.3f%6.2f%6.2f %-4s%2s\n",
397  recordname, indexbuf, atomname, altlocchar, resnamebuf, chain[0],
398  residbuf, insertion[0], x, y, z, occ, beta, segnamebuf, elementsymbol);
399 
400  return (rc > 0);
401 }
